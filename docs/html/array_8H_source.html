319  size_t msz = 1+((rowc*colc)/8);

327  }

329  if (this != &rhs) {

343  size_t i = (r*this->colc)+c;
344  size_t k = i / 8;

346  return (this->data[k] & (1 << b)) != 0;
347  }

349  size_t i = (r*this->colc)+c;

357  }
358  }
