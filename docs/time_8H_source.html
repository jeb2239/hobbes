24  str::pair u = str::readWhile(&str::isNotDigit, p.second);

29  } else if (u.first == "ms") { // milliseconds

32  m = 1000000;
33  } else if (u.first == "m" || u.first == "min") { // minutes
34  m = 1000000*60;
35  } else if (u.first == "h" || u.first == "hour") { // hours

38  m = 1000000L*60*60*24;

40
